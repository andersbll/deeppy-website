[docs]    def fprop(self, x, phase):
        if self.dropout > 0.0:
            if phase == 'train':
                self._tmp_mask = self.dropout < ca.random.uniform(size=x.shape)
                y = x * self._tmp_mask
            elif phase == 'test':
                y = x * (1.0 - self.dropout)
        return y
